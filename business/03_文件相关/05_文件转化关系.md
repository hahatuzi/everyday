# 转化关系：
  - files  --------new Blob--------->  Blobs  ---->  Append到formData   ----->  接口传输
  - Blobs  --------new File--------->  files  ---->  Append到formData   ----->  接口传输
  - files  ----->  FileReader  ------> base64
  - blobs  ----->  FileReader  ------> text  ----->  接口传输

# 转化代码
  ```js
    service.post(url, params, {
      headers: { "Content-Type": "application/json" },
      responseType: "blob",
    }).then((res) => {
      let { data } = res;
      const isBlob = blobValidate(data);
      if (isBlob) {
        const blob = new Blob([data]);
        let file = new File([blob], filename + ".xlsx", { type: blob.type });
        loadExcelAndRender(file, successCallback);

        // ================blob--->file==============================
        // 读取文件二进制数据信息
        const blob = res;
        // 将读取到的blob二进制数据作为一个新的File对象上传至另一个服务器端口
        const file = new File([blob], fileName, {
          type: blob.type,
        });
        const query = new FormData();
        query.append("file", file);
      }
    })
    .catch((r) => {
      console.error(r);
      ElMessage.error("获取数据失败");
    });
  ```
