# 4.declare
  - declare可以用来为已经存在的js库提供类型信息
  ```js
    declare function create(o:object | null):void;
    create({})
    import "vue-router";

    declare module "vue-router" {
      // https://router.vuejs.org/zh/guide/advanced/meta.html#typescript
      // 可以通过扩展 RouteMeta 接口来输入 meta 字段
      interface RouteMeta {
        /**
         * 菜单名称
         * @example 'Dashboard'
         */
        title?: string;

        /**
         * 菜单图标
         * @example 'el-icon-edit'
         */
        icon?: string;

        /**
         * 是否隐藏菜单项
         * true 隐藏, false 显示
         * @default false
         */
        hidden?: boolean;

        /**
         * 始终显示父级菜单，即使只有一个子菜单
         * true 显示父级菜单, false 隐藏父级菜单，显示唯一子节点
         * @default false
         */
        alwaysShow?: boolean;

        /**
         * 是否固定在页签上
         * true 固定, false 不固定
         * @default false
         */
        affix?: boolean;

        /**
         * 是否缓存页面
         * true 缓存, false 不缓存
         * @default false
         */
        keepAlive?: boolean;

        /**
         * 是否在面包屑导航中隐藏
         * true 隐藏, false 显示
         * @default false
         */
        breadcrumb?: boolean;

        /**
         * 拥有访问该菜单权限的角色编码集合
         * @example ['admin', 'editor']
         */
        roles?: string[];
      }
    }
  ```

# 6.类型保护:在运行ts代码时检查变量的类型，从而根据类型来执行不同的操作。通常使用instanceof和typeof实现
  ```js
    class Person {
      name:string;
      constructor (name:string) {
        this.name = name
      }
    }
    function greet(person:Person | string):void {
      if(person instanceof Person) {
        console.log('Hello,' + person.name)
      } else if (typeof person == 'string') {
        console.log('Hello,' + person)
      }
    }
    const lisa = new Person('lisa')
    greet(lisa)
    greet('lisa')
  ```

# 13.泛型工具
```js
  type Exclude<T, U> = T extends U ? never : T; // 强制T不是U的子类
  type Color = 'red' | 'green' | 'blue'
  type PrimaryColors = Exclude<Color, 'green' | 'red'>  // PrimaryColors是从Color颜色列表中排除'green' | 'red'的一种类型
  const primary: PrimaryColors = 'red'
  type Extract<T, U> = T extends U ? T : U; // 强制T是U的子类
  // 对属性列表K做类型映射
  type  Record<K extends keyof any, T> = {
    [P in K] : T;
  }
  type Partial<T> = {
    [  P  in  keyof  T ]?: T[P]
  }
    // 从T中节选属性列表K形成新的接口
  type Pick<T, K extends keyof T> = {
    [P in K]: T[P]
  }
  // 从T中忽略属性列表
  type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>
  // 模板文字类型
  type Greeting<T extends string> = `Hello, ${T}!`;
  type GreetJohn = Greeting<'John'>; // Result: "Hello, John!"
```
