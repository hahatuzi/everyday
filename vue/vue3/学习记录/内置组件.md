  ```js
    // transition触发条件：v-if,v-show,component切换的动态组件，key属性改变的组件
    <Transition>
      <div v-if="show"></div>
      <div v-show="show"></div>
      <component :is="defineAsyncComponent(() => import(`./comp/${curComponent.name}.vue`))"></component>
    </Transition>
    <TransitionGroup name="list" tag="ul">
      <li v-for="item in items" :key="item">
        {{ item }}
      </li>
    </TransitionGroup>
    ```
