# react路由的配置方式：分为两种：组件式和函数式
  ##  （1）组件式
  ```js
  function App() {
    return (
      <BrowserRouter basename="/app">
        <Routes>
          <Route path="/" /> {/* 👈 Renders at /app/ */}
        </Routes>
      </BrowserRouter>
    );
  }
  ```

  ##  （2）函数式
  1.通过createBrowserRouter 创建一个路由表
  ```js
    import  {lazy} from 'react'
    import {createBrowserRouter} from 'react-router-dom'

    const NotFound = lazy(() => import('@/views/404'))
    const Home = lazy(() => import('@/views/Home/index'))

    const router = createBrowserRouter([
      {path:'/', element:<Layout></Layout>, children:[{path:'/home',element: <Home></Home> },]}
      {path:'*',element:<NotFound></NotFound> },
    ])
    export default router
  ```
  2.然后通过RouterProvider，或者useRoutes()创建路由
  ```js
    // main.jsx文件
    import React from 'react'
    import ReactDOM from 'react-dom/client'
    import router from '@/router/data.jsx'
    import {RouterProvider } from "react-router-dom"

    ReactDOM.createRoot(document.getElementById('root')!).render(
      <React.StrictMode>
          <Suspense fallback={<div>正在加载中...</div>}>
          　<RouterProvider router={router} />
        　</Suspense>
      </React.StrictMode>
    )
  ```
# 路由嵌套：函数式路由配置下通过chidlren属性和<outlet>实现
  ```js
    // V6
    // Outlet用来表示嵌套路由中的组件
    // 当嵌套路由中的路径匹配成功时，Outlet表示嵌套路由中的组件
    // 当嵌套路由中的路径咩有匹配成功，Outlet就什么都不是
    <Routes>
      <Route path="user" element={<Users />}>
          {/* user/10 */}
          <Route path=":id" element={<UserDetail />} />
          {/* user/create */}
          <Route path="create" element={<NewUser />} /> 
      </Route>
    </Routes>
    function Users() {
      return (
        <div>
          <h1>Users</h1>
          <Outlet />
        </div>
      );
    }
    // 子组件的内容通过props.children来显示
    <Layout>
      <Routes>
        <Route path="user" element={<Users />}>
            <Route path=":id" element={<UserDetail />} />
            <Route path="create" element={<NewUser />} />
        </Route>
      </Routes>
    </Layout>
    function Layout(props) {
      return (
        <div>
          {props.chidlren}
        </div>
      );
    }
  ```
