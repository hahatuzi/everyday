# 一：定义
  ```js
    // 定义方法一：定义在constructor外部
    const App extends Component{
      state = {}
    }
    // 定义方法二：
    const App extends Component{
      constructor ()  {
        super()
        this.state = {
          username: 'lisa'
        }
      }
      function handleClick () {
        const { username } = this.state
        console.log(username)
        this.setState({username:'jisoo'})
      }
    }
  ```

# 二：事件绑定
  - (1)事件this指向
  ```js
    // (1)箭头函数:因为render函数的this指向类实例
    render () {  <div onClick={ () => {this.change()}}></div>}
    // (2)Function.prototype.bind()：将事件处理程序中的this直接与组件实例绑定在一起
    constructor () {
      this.change = this.change.bind(this)
    }
  ```
