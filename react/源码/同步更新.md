# 流程
  交互--> 将work插入到workList中 --> 取出work --> schedule  --> 调度work,交给perform执行 --> 继续调度
  ```js

  const workList = []
  button.onClick = () => {
    workList.unshift({count:100})
    schedule()
  }
    function schedule () {
      const curWork = workList.pop()
      if(curWork) {
        perform(curWork)
      }
    }

    function perform (work:Work) {
      while(work.count){
        work.count--
        insertSpan('0')
      }
      schedule()
    }

    function insertSpan(content){
      const span = document.createElement('span')
      span.innerText = content
      root?.appendChild(span)
    }
  ```
