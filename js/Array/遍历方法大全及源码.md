# forEach
  ```js
    Array.prototype.myForEach = function (fn,thisValue) {
      // console.log(this)
      // if (!thisValue) thisValue = undefined
      for(var i = 0;  i<this.length;i++){
        //此处的this[I]就是我们用来判断myForEach是否发生改变的依据
        fn(this[i])
      }
    }
    var arr2 = [1,4,2]
    arr2.myForEach(item=>{item = item*2})
    console.log(arr2)
    var arr2 = [{age:12},{age:10}]
    arr2.myForEach(item=>{item.age = item.age*2})
    console.log(arr2)
  ```
